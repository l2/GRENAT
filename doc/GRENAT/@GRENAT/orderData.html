0029 function dataOut=orderData(obj,dataIn,type)
0030 switch type
0031     case 'sampleIn'
0032         %size of the input data
0033         obj.sizeNonSample=[size(dataIn,1),size(dataIn,2),size(dataIn,3)];
0034         %in the case of nd-array
0035         if obj.sizeNonSample(3)>1
0036             dataOut=reshape(dataIn,[ obj.sizeNonSample(1)*obj.sizeNonSample(2),obj.sizeNonSample(3),1]);
0037         else
0038             dataOut=dataIn;
0039         end
0040     case 'sampleOut'
0041         if obj.sizeNonSample(3)>1

0044             dataOut=dataIn;
0045         end
