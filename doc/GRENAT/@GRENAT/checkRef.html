0031 okSample=false;

0043 if sum(nbSRef(:))~=0
0044     okSample=true;
0045     if nbSRef(1)==nbRRef(1)
0046         okResp=true;
0047     end
0048     if nbGRef(3)==1
0049         if nbGRef(1)==nbSRef(1)&&nbGRef(2)==nbSRef(2)
0050             okGrad=true;
0051         end
0052     elseif nbGRef(3)==nbSRef(2)&&nbGRef(1)==nbSRef(1)
0053         okGrad=true;
0054     elseif all(nbGRef==nbSRef)
0055         okGrad=true;
0056     end
0057 end
0058 okAll=okSample&&okResp&&okGrad;
0059 %display error messages

0061 if ~okResp;Gfprintf('>> Wrong definition of the reference responses\n');end
