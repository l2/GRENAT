0031 function [r,radj,r2,r2adj,rccc]=corrFact(Zex,Zap)
0032 
0033 Zex=Zex(:);Zap=Zap(:);
0034 nv=length(Zex);
0035 % computation of the empirical means
0036 meanZex=mean(Zex);
0037 meanZap=mean(Zap);
0038 
0039 % computation of the empirical covariance
0040 covZexZap=1/nv*sum((Zex-meanZex).*(Zap-meanZap));
0041 
0042 % computation of the empirical variances
0043 VZex=1/nv*sum((Zex-meanZex).^2);
0044 VZap=1/nv*sum((Zap-meanZap).^2);
0045 
0046 % Pearson's correlation coefficient
0047 r=covZexZap/sqrt(VZex*VZap);
0048 %squared correlation coefficient
0049 r2=r^2;
0050 %adjusted correlation coefficinet
0051 radj=sqrt(1-(nv-1)/(nv-2)*(1-r2));
0052 %adjusted squared correlation coefficinet
0053 r2adj=radj^2;
0054 
0055 %Concordance Correlation Coefficient
0056 rccc=2*covZexZap/(VZex+VZex+(meanZex-meanZap)^2);
