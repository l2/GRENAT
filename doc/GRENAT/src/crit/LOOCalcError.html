0021 function [ret]=LOOCalcError(Zref,Zap,variance,GZref,GZap,ns,np,LOO_norm)
0022 
0023 %no norm specified
0024 if nargin<8
0025     LOO_norm='L2';
0026 end
0027 
0028 %check availability of the gradients
0029 availGrad=true;
0030 if isempty(GZref)
0031     availGrad=false;
0032 end
0033 %check calculation of SCVR
0034 availVar=true;
0035 if isempty(variance)
0036     availVar=false;
0037 end
0038 
0039 %diff responses
0040 diffZ=Zap-Zref;
0041 if availGrad
0042     %diff gradients
0043     diffG=GZap-GZref;
0044 end
0045 %diff responses (choice of the norm)
0046 switch LOO_norm
0047     case 'L1'
0048         diffC=abs(diffZ);
0049     case 'L2'
0050         diffC=diffZ.^2;
0051     case 'Linf'
0052         diffC=max(diffZ(:));
0053 end
0054 %Custom criterion
0055 somm=0.5*(Zap+Zref);
0056 ret.errp=1/ns*sum(abs(diffZ)./somm);
0057 %PRESS
0058 ret.press=sum(diffC);
0059 %mean bias
0060 ret.bm=1/ns*sum(diffZ);
0061 if availGrad
0062     %diff gradients (choice of the norm)
0063     switch LOO_norm
0064         case 'L1'
0065             diffgc=abs(diffG);
0066         case 'L2'
0067             diffgc=diffZ.^2;
0068         case 'Linf'
0069             diffgc=max(diffG);
0070     end
0071     %mean of the differences on responsesn gradients and both squared
0072     ret.eloor=1/ns*sum(diffC);
0073     ret.eloog=1/(ns*np)*sum(diffgc(:));
0074     ret.eloot=1/(ns*(1+np))*(sum(diffC)+sum(diffgc(:)));
0075 else
0076     %mean diff responses
0077     ret.eloor=1/ns*sum(diffC);
0078     ret.eloot=ret.eloor;
0079 end
0080 if availVar
0081     %criterion of adequation (SCVR Keane 2005/Jones 1998)
0082     ret.scvr=diffZ./variance;
0083     ret.scvr_min=min(ret.scvr(:));
0084     ret.scvr_max=max(ret.scvr(:));
0085     ret.scvr_mean=mean(ret.scvr(:));
0086     %%criterion of adequation (CAUTION of the norm!!!>> squared difference)
0087     diffA=diffC./variance;
0088     ret.adequ=1/ns*sum(diffA);
0089 end
