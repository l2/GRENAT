0021 function [ret]=LOOCalcError(Zref,Zap,variance,GZref,GZap,ns,np,LOO_norm)
0022 
0023 %check availability of the gradients
0024 availGrad=true;
0025 if isempty(GZref)
0026     availGrad=false;
0027 end
0028 %check calculation of SCVR
0029 availVar=true;
0030 if isempty(variance)
0031     availVar=false;
0032 end

0034 %diff responses
0035 diffZ=Zap-Zref;
0036 if availGrad
0037     %diff gradients
0038     diffG=GZap-GZref;
0039 end
0040 %diff responses (choice of the norm)
0041 switch LOO_norm
0042     case 'L1'
0043         diffC=abs(diffZ);
0044     case 'L2'
0045         diffC=diffZ.^2;
0046     case 'Linf'
0047         diffC=max(diffZ(:));
0048 end
0049 %Custom criterion
0050 somm=0.5*(Zap+Zref);
0051 ret.errp=1/ns*sum(abs(diffZ)./somm);
0052 %PRESS
0053 ret.press=sum(diffC);
0054 %mean bias
0055 ret.bm=1/ns*sum(diffZ);
0056 if availGrad
0057     %diff gradients (choice of the norm)
0058     switch LOO_norm
0059         case 'L1'
0060             diffgc=abs(diffG);
0061         case 'L2'
0062             diffgc=diffZ.^2;
0063         case 'Linf'
0064             diffgc=max(diffG);
0065     end
0066     %mean of the differences on responsesn gradients and both squared
0067     ret.eloor=1/ns*sum(diffC);
0068     ret.eloog=1/(ns*np)*sum(diffgc(:));
0069     ret.eloot=1/(ns*(1+np))*(sum(diffC)+sum(diffgc(:)));
0070 else
0071     %mean diff responses

0073     ret.eloot=ret.eloor;
0074 end

0077     ret.scvr=diffZ./variance;

0079     ret.scvr_max=max(ret.scvr(:));
0080     ret.scvr_mean=mean(ret.scvr(:));
0081     %%criterion of adequation (CAUTION of the norm!!!>> squared difference)
0082     diffA=diffC./variance;
