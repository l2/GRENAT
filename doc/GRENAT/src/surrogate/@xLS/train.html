0029 if nargin==1;flagRun=true;end

0035 if ~obj.flagG
0036     obj.valFunPoly=multiMono(obj.sampling,obj.funPoly);
0037     if obj.checkMiss
0038         %remove missing response(s)
0039         obj.valFunPoly=obj.missData.removeRV(obj.valFunPoly);
0040     end
0041 else
0042     %gradient-based
0043     [MatX,MatDX]=multiMono(obj.sampling,obj.funPoly);
0044     %remove lines associated to the missing data
0045     if obj.checkMiss
0046         obj.valFunPoly=obj.missData.removeRV(MatX);
0047         obj.valFunPolyD=obj.missData.removeGV(MatDX);
0048     else
0049         obj.valFunPoly=MatX;

0051     end
0052 end
