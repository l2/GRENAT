0024 function [logLi,Li,liSack]=KRGLikelihood(dataKRG)
0025 
0026 %size of the kernel matrix
0027 sizeK=size(dataKRG.build.KK,1);
0028 
0029 %computation of the log-likelihood (Jones 1993 / Leary 2004)
0030 switch dataKRG.build.factKK
0031     case 'QR'
0032         diagRK=diag(dataKRG.build.RK);
0033         detK=abs(prod(diagRK)); %Q is an unitary matrix
0034         logDetK=sum(log(abs(diagRK)));
0035     case 'LL'
0036         diagLK=diag(dataKRG.build.LK);
0037         detK=prod(diagLK)^2;
0038         logDetK=2*sum(log(abs(diagLK)));
0039     case 'LU'
0040         diagUK=diag(dataKRG.build.UK);
0041         detK=prod(diagUK); %L is a quasi-triangular matrix and contains ones on the diagonal
0042         logDetK=sum(log(abs(diagUK)));
0043     otherwise
0044         eigVal=eig(dataKRG.build.K);
0045         detK=prod(eigVal);
0046         logDetK=sum(log(eigVal));
0047 end
0048 
0049 logLi=sizeK/2*log(2*pi*dataKRG.build.sig2)+1/2*logDetK+sizeK/2;
0050 liSack=0;
0051 if nargout==3
0052     liSack=abs(detK)^(1/sizeK)*dataKRG.build.sig2;
0053 end
0054 
0055 
0056 if isinf(logLi)||isnan(logLi)
0057     logLi=1e16;
0058 end
0059 
0060 if nargout>=2
0061     %computation of the likelihood (Jones 1993 / Leary 2004)
0062     Li=1/((2*pi*dataKRG.build.sig2)^(sizeK/2)*sqrt(detK))*exp(-sizeK/2);
0063     
0064 elseif nargout >3
0065     error(['Wrong number of output variables (',mfilename,')\n']);
0066 end
