0025 % - ff: monomial evaluated at the evaluation points
0026 % OUTPUTS:
0027 % - variance: value of the kriging variance

0029 function variance=computeVariance(obj,rr,ff)
0030 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
0031 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
0032 %compute the prediction variance (MSE) (Lophaven, Nielsen & Sondergaard
0033 %2004 / Marcelet 2008 / Chauvet 1999)
0034 %
0035 %depending on the factorization
0036 switch obj.factK
0037     case 'QR'
0038         rrP=rr'*obj.matrices.PK;
0039         Qrr=obj.matrices.QtK*rr;
0040         u=obj.matrices.fcK*Qrr-ff';
0041         variance=obj.sig2*(1-(rrP/obj.matrices.RK)*Qrr+...
0042             u'/obj.matrices.fcCfct*u);
0043     case 'LU'
0044         rrP=rr(obj.matrices.PK,:);
0045         Lrr=obj.matrices.LK\rrP;
0046         u=obj.matrices.fcU*Lrr-ff';
0047         variance=obj.sig2*(1-(rr'/obj.matrices.UK)*Lrr+...
0048             u'/obj.matrices.fcCfct*u);
0049     case 'LL'
0050         Lrr=obj.matrices.LK\rr;
0051         u=obj.matrices.fcL*Lrr-ff';
0052         variance=obj.sig2*(1-(rr'/obj.matrices.LK')*Lrr+...
0053             u'/obj.matrices.fcCfct*u);
0054     otherwise
0055         rKrr=obj.matrices.KK \ rr;
0056         u=obj.matrices.fcC*rKrr-ff';
0057         variance=obj.sig2*(1+u'/obj.matrices.fcCfct*u - rr'*rKrr);
0058 end
0059 end
