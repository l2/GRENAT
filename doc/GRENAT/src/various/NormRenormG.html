0021 function [out]=NormRenormG(in,type,infoDataS,infoDataR)
0022 
0023 % number of sample points
0024 ns=size(in,1);
0025 % normalisation of the data
0026 if (nargin>=3&&~isempty(infoDataS.std))||nargin==2
0027     switch type
0028         case 'norm'
0029             stdS=infoDataS.std;
0030             stdR=infoDataR.std;
0031             out=in.*stdS(ones(ns,1),:)./stdR;
0032         case 'renorm'
0033             stdS=infoDataS.std;
0034             stdR=infoDataR.std;
0035             out=in*stdR./stdS(ones(ns,1),:);
0036         case 'renorm_concat'  %concatenated gradients in a vector
0037             stdS=infoDataS.std;
0038             stdR=infoDataR.std;
0039             correct=stdR./stdS;
0040             nbv=numel(stdS);
0041             out=in.*repmat(correct(:),ns/nbv,1);
0042         otherwise
0043             Gfprintf('Wrong kind of normalisation/renormalisation');
0044             error(['Error in ' mfilename ]);
0045     end

0047     out=in;
