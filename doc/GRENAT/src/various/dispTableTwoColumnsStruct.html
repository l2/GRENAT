0027 function dispTableTwoColumnsStruct(tableFieldIn,structIn)
0028 %size of every components in tableA
0029 sizeA=cellfun(@numel,tableFieldIn);
0030 maxA=max(sizeA);
0031 %space after each component
0032 spaceA=maxA-sizeA+3;
0033 spaceTxt=' ';
0034 %display table
0035 for itT=1:numel(tableFieldIn)
0036     if isfield(structIn,tableFieldIn{itT})
0037         fprintf('%s%s%s\n',tableFieldIn{itT},spaceTxt(ones(1,spaceA(itT))),structIn.(tableFieldIn{itT}));
0038     end
0039 end
