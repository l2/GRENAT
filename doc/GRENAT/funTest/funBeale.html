0035     dem=true;

0037 
0038 if ~isempty(xx)
0039     if size(xx,3)>2
0040         error('The Beale's function is a 2 dimensional function');
0041     elseif size(xx,3)==1
0042         if size(xx,2)==2
0043             xxx=xx(:,1);yyy=xx(:,2);
0044         elseif size(xx,1)==2
0045             xxx=xx(:,2);yyy=xx(:,1);

0052     end

0054     p = (1.5 - xxx + xxx.*yyy).^2 + (2.25 - xxx + xxx.*yyy.^2).^2 + (2.625 - xxx + xxx.*yyy.^3).^2;
0055

0057     if nargout==2||dem
0058         dp(:,:,1)=2*(yyy-1).*(1.5-xxx+xxx.*yyy)+...
