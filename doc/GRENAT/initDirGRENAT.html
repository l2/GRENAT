0119 %check if a directory exists in the path or not and add it to the path if not
0120 function flag=addpathExisted(folder)
0121 flag=1;
0122 folder=fullfile(folder);
0123 if ispc
0124     % Windows is not case-sensitive
0125     onPath = contains(lower(path),lower(folder));
0126 else
0127     onPath = contains(path,folder);
0128 end
0129 %check of the folder contains @ (if yes it will not be added to the path)
0130 inAt=contains(folder,'@');
0131 if exist(folder,'dir')&&~onPath&&~inAt
0132     flag=2;
